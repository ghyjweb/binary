5. 代码实现 (C++)

binarySearch.cpp
int binarySearch(int arr[], int size, int target) {
int left = 0, right = size - 1; // 1. 初始化边界
while (left <= right) { // 只要搜索范围有效
int mid = left + (right - left) / 2; // 2. 防溢出求中点
if (arr[mid] == target) {
return mid; // 3. 匹配成功，返回索引
}
else if (arr[mid] < target) {
left = mid + 1; // 目标在右侧，左界右移
}
else {
right = mid - 1; // 目标在左侧，右界左移
}
}
return -1; // 未找到目标，返回-1
}
